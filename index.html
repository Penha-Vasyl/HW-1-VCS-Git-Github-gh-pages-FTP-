Шпаргалка з командами git

git config --global user.name Penha Vasul - налаштовує ім'я користувача;
git config --global user.email vasylpenha@gmail.com - налаштовує емейл користувача;
git init - створює папку на ком'ютері - новий, локальний репозиторій;
git status - показує, всі нові або змінені файли, які не збережені в git;
git add - додає збережений файл, для збереження в комміт;
git commit - збергіає нові файли і зміни в комміт, тобто історію збережень, яку потім можна відновити з цієї "точки" збереження;
git log - показує історію збереження коммітів, дату коли збережено і хто зберіг;
git checkout - переключає на іншу, обрану гілку, і оновляє дані і файли в папці на локальному репозиторію;
git branch - видаляє обрану гілку;
git push - загружає всі збережені зміни з локального репозиторію на віддалений репозиторій;
git pull - завантажує історію збережень з віддаленого репозиторію, і обєднує їх з локальним;
git merge - робить злиття додаткової гілки з основною, зберігає хронологію комітів і створює новий, вже злитий коміт;
git clone -скачує репозиторій, зі всіма змінами, які відбувались;
.gitignore файл - вказуються списокок файлів і папок, які git ігноруватиvt;
git revert - дія яка затирває попередні зміни, але при тому створює новий коміт, з затертими змінами;
git reset - видаляє обрані файли з індексу;
git rebase - робить злиття комітів між гілками, додаючи всі коміти з гілки в кінець основної гілки, роблячи зміни в комітах;
git cherry-pick - дає можливість злити в один, обрані коміти;
git stash - збергіє змінити, які не додаються в коміт, але за потреби можна відновити ці збереження;
